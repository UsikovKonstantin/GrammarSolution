# После "start:" нужно записать символ, с которого начинается составление цепочек
# Дальше нужно описать правила грамматики
# Для обозначения пустого слова использовать "_"

# Пример
start: S	# Начинаем с "S"
S: AAA B	# Из "S" можно получить "AAA" или "B"
A: aA B		# Из "A" можно получить "aA" или "B"
B: _		# Из "B" можно получить пустое слово
